 37def allow_memory_growth() -> None:
 38    """Allow memory growth for TensorFlow GPU devices."""
 39    # Allow memory growth
 40    # Only in main process
 41    import tensorflow as tf
 42
 43    if gpus := tf.config.list_physical_devices("GPU"):
 44        try:
 45            for gpu in gpus:
 46                tf.config.experimental.set_memory_growth(gpu, True)
 47        except RuntimeError:
 48            logger.exception("Failed to set memory growth")

 97def batch_tensor_slices(
 98    data: BatchableT, batch_size: int, desc: str | None = None, verbose: int = 0
 99) -> Iterable[BatchableT]:
100    """Batch tensor slices to :class:`tf.data.Dataset`.
101
102    Args:
103        data: Data to batch. Either a single tensor or a tuple! of tensors.
104        batch_size: Batch size.
105        desc: Description for tqdm. Defaults to None.
106        verbose: Verbosity level. Defaults to 0.
107
108    Returns:
109        A iterable over a :class:`tf.data.Dataset` of batched data.
110    """
111    import tensorflow as tf
112
113    dataset = (
114        tf.data.Dataset.from_tensor_slices(data)
115        .batch(batch_size)
116        .prefetch(tf.data.AUTOTUNE)
117    )
118    if verbose > 0:
119        batches = ceil(len(data) / batch_size)
120        return tqdm(dataset, desc=desc, total=batches)  # type: ignore[no-any-return]
121
122    return dataset  # type: ignore[no-any-return]

160@tf_func_wrapper
161def lstm_model_call(
162    layers: Sequence,
163    lstm_dims: Sequence[int],
164    x_t: tf.Tensor,
165    hidden_states: Sequence[Sequence[tf.Tensor]],
166    training: bool = False,
167) -> tuple[tf.Tensor, list[list[tf.Tensor]], None]:
168    """Call a model with LSTM layers for one time step (tf.function wrapped).
169
170    Args:
171        layers: Model layers.
172        lstm_dims: Neurons per LSTM layer.
173        x_t: Input at time step t.
174            If embedding: [Batch] {int32}
175            Else: [Batch, Vocab] {float32}
176        hidden_states: LSTM layers' hidden states. [2 x [Batch, Hidden] {float32}, ...]
177        training: Whether in training mode. Defaults to False.
178
179    Returns:
180        tuple[tf.Tensor, list[list[tf.Tensor]], None]:
181            output: Time step t output. [Batch, Vocab] {float32}
182            states: LSTM layers' output states. [2 x [Batch, Hidden] {float32}, ...]
183            info: Always None.
184    """
185    import tensorflow as tf
186
187    # Set initial LSTM index to 0
188    lstm_ix = 0
189
190    # Create list for output states
191    output_states: list[list[tf.Tensor]] = [[] for _ in lstm_dims]
192
193    # Loop through layers in the model
194    for layer in layers:
195        # Apply batch normalization or time distributed layer
196        # Expand, apply, squeeze as these layers expect 3D input
197        if isinstance(
198            layer, tf.keras.layers.BatchNormalization | tf.keras.layers.TimeDistributed
199        ):
200            x_t = tf.expand_dims(x_t, axis=1)
201            x_t = layer(x_t, training=training)
202            x_t = tf.squeeze(x_t, axis=1)
203
204        # Apply LSTM layer
205        elif isinstance(layer, tf.keras.layers.LSTM):
206            x_t, curr_states = layer.cell(
207                x_t, hidden_states[lstm_ix], training=training
208            )
209
210            output_states[lstm_ix] = curr_states
211            lstm_ix += 1
212
213        # Apply other layer types
214        else:
215            x_t = layer(x_t, training=training)
216
217    # Return output and output states, and None for compatibility with subclasses
218    return x_t, output_states, None

285def plot_history(history_data: HistoryProto | dict[str, Sequence[float]]) -> Figure:
286    """Plot training history to file.
287
288    Calculates the best epoch and loss from the validation data.
289
290    Args:
291        history_data: Keras history object or dictionary of metrics
292
293    Returns:
294        Image object
295    """
296    # - include a scaled LR plot, if LR is available in the history
297    # - accept dictionary of metrics to plot
298    history: dict[str, Sequence[float]] = (
299        history_data.history  # type: ignore[assignment]
300        if isinstance(history_data, HistoryProto)
301        else history_data
302    )
303
304    fig = Figure(figsize=(10, 6))
305    ax: Axes = fig.add_subplot(111)
306    losses = history["loss"]
307    val_losses = history["val_loss"]
308    ax.plot(losses, label="loss")
309    ax.plot(val_losses, label="val_loss")
310    if "lr" in history:
311        lrs = history["lr"]
312        ax2: Axes = ax.twinx()
313        ax2.plot(lrs, color="C2")
314        ax.plot([], color="C2", label="lr")
315        ax2.set_ylabel("lr")
316
317    best_epoch = int(np.argmin(val_losses)) + 1
318    best_loss = min(val_losses)
319    ax.set_title(f"Best epoch: {best_epoch}, val_loss: {best_loss:.3f}")
320    ax.legend()
321    fig.tight_layout()
322    return fig
